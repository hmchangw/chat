# Loadgen v2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement loadgen v2 as scoped in `docs/superpowers/specs/2026-05-12-loadgen-v2-design.md` — trustworthy verdicts, restructured internals, and chat-shaped load coverage.

**Architecture:** Five phases (0 → 1a → 1b → 2 → 3) plus a cross-phase docs/ops cohort. Phase 0 is a pure refactor that extracts `Runtime`, `runFlags`, and transport adapters before any behavior change so Phase 1 lands on a clean substrate. Phase 1a adds the trust-core primitives (HDR histograms, coordinated-omission tracking, RUN QUALITY verdict, settle phase). Phase 1b layers operability (artifact bundles, run isolation, teardown-force, alerts, dashboards, security, soak). Phase 2 replaces per-scenario string switches with a registered `Scenario` interface family. Phase 3 adds 14 chat-shaped scenarios (RAW timing, large-room ×3 shapes, presence/typing, notifications, threads, edits/deletes, subscription churn, auth, federation with 4 sub-stages + flap + cross-read, DM/channel split, first-DM, room-open composite, bursty publish, read receipts, chaos overlay).

**Tech Stack:** Go 1.25 · `HdrHistogram-go` (with t-digest fallback if dep declined) · `nats.go` + `jetstream` · MongoDB v2 driver · Cassandra (gocql) · Resty · Gin · `caarlos0/env` · `go.uber.org/mock` · `testify` · `testcontainers-go` · `toxiproxy` for chaos overlay · Prometheus + Grafana · Docker Compose v1 (≥1.29.2) or v2 (≥2.0).

**Process discipline (from `CLAUDE.md`):**
- All new code follows Red-Green-Refactor TDD. Never write implementation before a failing test exists.
- Always use `make` targets (`make lint`, `make test SERVICE=tools/loadgen`, `make test-integration SERVICE=tools/loadgen`, `make generate SERVICE=tools/loadgen`).
- Wrap errors with `fmt.Errorf("doing X: %w", err)`, never bare `err`.
- Use `log/slog` JSON for all logging; structured fields, no interpolated strings.
- Update `tools/loadgen/USAGE.md` in the same PR as any operator-visible surface change.
- Lint and tests are pre-commit-hook enforced; fix failures before retrying.
- One commit per task. Push to branch `claude/loadgen-v2-impl` (or per-phase branches if you prefer).

**Approval gate before Phase 1a:** The HDR-histogram third-party dependency (`github.com/HdrHistogram/hdrhistogram-go`) requires owner sign-off per CLAUDE.md §5. package main

```go
import (
    "context"
    "encoding/json"
    "fmt"
    "log/slog"

    "github.com/nats-io/nats.go"
    "github.com/nats-io/nats.go/jetstream"
    "hmchangw/chat/pkg/model"
)

const HeaderRunID = "X-Loadgen-Run-ID"

type Publisher interface {
    Publish(ctx context.Context, subject string, body []byte) error
}

type natsCorePublisher struct {
    nc          *nats.Conn
    js          jetstream.JetStream
    useJetStream bool
    asyncJS     bool
    runID       string
}

func (p *natsCorePublisher) Publish(ctx context.Context, subject string, body []byte) error {
    msg := &nats.Msg{Subject: subject, Data: body, Header: nats.Header{HeaderRunID: []string{p.runID}}}
    switch {
    case p.useJetStream && p.asyncJS:
        if _, err := p.js.PublishMsgAsync(msg); err != nil {
            return fmt.Errorf("jetstream publish async: %w", err)
        }
    case p.useJetStream:
        if _, err := p.js.PublishMsg(ctx, msg); err != nil {
            return fmt.Errorf("jetstream publish: %w", err)
        }
    default:
        if err := p.nc.PublishMsg(msg); err != nil {
            return fmt.Errorf("nats publish: %w", err)
        }
    }
    return nil
}

func newAsyncErrHandler(c *Collector, presetName string) jetstream.MsgErrHandler {
    return func(_ jetstream.JetStream, m *nats.Msg, err error) {
        var evt model.CanonicalCreatedEvent
        if jerr := json.Unmarshal(m.Data, &evt); jerr != nil {
            slog.Warn("async ack handler could not parse payload",
                slog.String("error", jerr.Error()),
                slog.String("preset", presetName))
            return
        }
        c.RecordPublishFailed("", evt.Message.ID)
        slog.Warn("async jetstream publish nacked",
            slog.String("messageID", evt.Message.ID),
            slog.String("error", err.Error()))
    }
}

func jetstreamPublishOpts(maxPending int) []jetstream.PublishOpt {
    return []jetstream.PublishOpt{
        jetstream.WithRetryAttempts(0), // open-loop, do not block the generator
    }
}
```

```go
import (
    "context"

    "github.com/nats-io/nats.go"

type Requester interface {
    Request(ctx context.Context, subject string, body []byte, deadline time.Duration) ([]byte, error)
}

type natsRequester struct {
    pool  *ConnPool
    runID string
}

func (r *natsRequester) Request(ctx context.Context, subject string, body []byte, deadline time.Duration) ([]byte, error) {
    nc := r.pool.For(subject)
    msg := &nats.Msg{Subject: subject, Data: body, Header: nats.Header{HeaderRunID: []string{r.runID}}}
    reply, err := nc.RequestMsgWithContext(ctx, msg)
    if err != nil {
        return nil, fmt.Errorf("nats request %s: %w", subject, err)
    }
    return reply.Data, nil
}
```

```go
package main

import (
    "testing"
    "github.com/prometheus/client_golang/prometheus"
    dto "github.com/prometheus/client_model/go"
)

func gatheredCounterValue(t *testing.T, reg *prometheus.Registry, name string, labelMatch map[string]string) float64 {
    t.Helper()
    families, err := reg.Gather()
    if err != nil { t.Fatalf("gather: %v", err) }
    for _, f := range families {
        if f.GetName() != name { continue }
        for _, m := range f.Metric {
            if !labelsMatch(m.Label, labelMatch) { continue }
            if m.Counter != nil { return m.Counter.GetValue() }
        }
    }
    return 0
}

func counterValue(t *testing.T, reg *prometheus.Registry, name string) float64 {
    return gatheredCounterValue(t, reg, name, nil)
}

func counterValueLabeled(t *testing.T, reg *prometheus.Registry, name string, labels map[string]string) float64 {
    return gatheredCounterValue(t, reg, name, labels)
}

func labelsMatch(have []*dto.LabelPair, want map[string]string) bool {
    if len(want) == 0 { return true }
    for k, v := range want {
        found := false
        for _, lp := range have {
            if lp.GetName() == k && lp.GetValue() == v { found = true; break }
        }
        if !found { return false }
    }
    return true
}
```

```go
package main

import (
    "context"
    "fmt"
    "log/slog"
    "net/http"
    "os"
    "path/filepath"

    "github.com/nats-io/nats.go"
    "github.com/nats-io/nats.go/jetstream"
    "github.com/prometheus/client_golang/prometheus"
)

type Runtime struct {
    cfg        config
    runID      string
    nc         *nats.Conn
    js         jetstream.JetStream
    pool       *ConnPool
    metrics    *Metrics
    collector  *Collector
    metricsSrv *http.Server
    pprofSrv   *http.Server
    samplers   []sampler
    publisher  *natsCorePublisher
    requester  *natsRequester
}

func NewRuntime(ctx context.Context, cfg config, runID string) (*Runtime, error) {
    nc, err := nats.Connect(cfg.NATSURL)
    if err != nil { return nil, fmt.Errorf("connect NATS: %w", err) }
    js, err := jetstream.New(nc)
    if err != nil { return nil, fmt.Errorf("init jetstream: %w", err) }
    pool, err := NewConnPoolWithObserver(cfg.NATSURL, cfg.Connections, nc)
    if err != nil { return nil, fmt.Errorf("build conn pool: %w", err) }
    reg := prometheus.NewRegistry()
    metrics := NewMetrics(reg)
    collector := NewCollector()
    rt := &Runtime{
        cfg: cfg, runID: runID,
        nc: nc, js: js, pool: pool,
        metrics: metrics, collector: collector,
    }
    rt.publisher = &natsCorePublisher{nc: nc, js: js, runID: runID}
    rt.requester = &natsRequester{pool: pool, runID: runID}
    rt.metricsSrv = startMetricsServer(cfg.MetricsAddr, reg)
    if cfg.PprofAddr != "" {
        rt.pprofSrv = startPprofServer(cfg.PprofAddr)
    }
    return rt, nil
}
```

```go
package main

import (
    "fmt"
    "time"
    hdr "github.com/HdrHistogram/hdrhistogram-go"
)

// CellHistogram backs per-(scenario, kind, phase) latency cells.
// Floor 100µs, ceiling 60s, 3 sig digits. ~30 KB per histogram.
type CellHistogram struct{ h *hdr.Histogram }

func NewCellHistogram() *CellHistogram {
    return &CellHistogram{h: hdr.New(int64(100*time.Microsecond), int64(60*time.Second), 3)}
}

func (c *CellHistogram) Record(d time.Duration) {
    _ = c.h.RecordValue(int64(d))
}

func (c *CellHistogram) Quantile(q float64) time.Duration {
    return time.Duration(c.h.ValueAtQuantile(q))
}

func (c *CellHistogram) Count() int64 { return c.h.TotalCount() }

func (c *CellHistogram) Merge(other *CellHistogram) error {
    if dropped := c.h.Merge(other.h); dropped > 0 {
        return fmt.Errorf("merging histograms: %d values dropped (out of range)", dropped)
    }
    return nil
}

// Reset clears the histogram (used when a run starts a new measured-window phase).
func (c *CellHistogram) Reset() { c.h.Reset() }

// Export serializes the histogram in HDR log format for inclusion in
// the artifact bundle's histograms.hlog (Phase 1b §1.8).
func (c *CellHistogram) Export() ([]byte, error) {
    return c.h.Export()
}

// WindowHistogram backs the abort-watcher rolling window.
// Tighter floor 10µs since the watcher is safety-critical.
type WindowHistogram struct{ h *hdr.Histogram }

func NewWindowHistogram() *WindowHistogram {
    return &WindowHistogram{h: hdr.New(int64(10*time.Microsecond), int64(60*time.Second), 3)}
}

func (w *WindowHistogram) Record(d time.Duration) { _ = w.h.RecordValue(int64(d)) }
func (w *WindowHistogram) Quantile(q float64) time.Duration { return time.Duration(w.h.ValueAtQuantile(q)) }
func (w *WindowHistogram) Count() int64 { return w.h.TotalCount() }
func (w *WindowHistogram) Reset() { w.h.Reset() }
```

```go
import (
    "time"
```
```go
    "github.com/prometheus/client_golang/prometheus"
```

```go
type OmissionTracker struct {
    serviced *CellHistogram
    dropped  *CellHistogram
    m        *Metrics
}

func NewOmissionTracker(m *Metrics) *OmissionTracker {
    return &OmissionTracker{
        serviced: NewCellHistogram(),
        dropped:  NewCellHistogram(),
        m:        m,
    }
}

func (o *OmissionTracker) RecordServiced(intendedAt, actualStartAt time.Time) {
    d := actualStartAt.Sub(intendedAt)
    if d < 0 { d = 0 }
    o.serviced.Record(d)
    o.m.OmissionDeficit.With(prometheus.Labels{"dropped": "false"}).Observe(d.Seconds())
}

func (o *OmissionTracker) RecordDropped(intendedAt, droppedAt time.Time) {
    d := droppedAt.Sub(intendedAt)
    if d < 0 { d = 0 }
    o.dropped.Record(d)
    o.m.OmissionDeficit.With(prometheus.Labels{"dropped": "true"}).Observe(d.Seconds())
}

func (o *OmissionTracker) Quantile(q float64, dropped bool) time.Duration {
    if dropped { return o.dropped.Quantile(q) }
    return o.serviced.Quantile(q)
}
```

```go
package main

import (
    "context"
    "fmt"
    "time"
)

type SettleFlags struct {
    Timeout  time.Duration
    Interval time.Duration
    Probes   int // N message-IDs to sample
}

type SettleOutcome struct {
    AllSucceeded bool
    Succeeded    int
    Failed       int
    Probes       []string
}

type ProbeFn func(ctx context.Context, id string) error

func Settle(ctx context.Context, sf SettleFlags, sampleIDs []string, probe ProbeFn) (SettleOutcome, error) {
    out := SettleOutcome{Probes: sampleIDs}
    if sf.Probes == 0 || len(sampleIDs) == 0 {
        out.AllSucceeded = true
        return out, nil
    }
    deadline := time.Now().Add(sf.Timeout)
    ok := make(map[string]bool)
    for time.Now().Before(deadline) && len(ok) < len(sampleIDs) {
        for _, id := range sampleIDs {
            if ok[id] { continue }
            if err := probe(ctx, id); err == nil {
                ok[id] = true
            }
        }
        if len(ok) >= len(sampleIDs) { break }
        select {
        case <-ctx.Done():
            return out, fmt.Errorf("settle aborted: %w", ctx.Err())
        case <-time.After(sf.Interval):
        }
    }
    out.Succeeded = len(ok)
    out.Failed = len(sampleIDs) - len(ok)
    out.AllSucceeded = (out.Failed == 0)
    if !out.AllSucceeded {
        return out, fmt.Errorf("settle timed out: %d/%d probes succeeded", out.Succeeded, len(sampleIDs))
    }
    return out, nil
}
```

```go
package main

import (
    "fmt"
    "strings"
    "time"
)

type RunQualityInputs struct {
    DrainTimedOut        bool
    MeasuredDuration     time.Duration
    AbortP99Sustain      time.Duration
    WarmupErrorRate      float64
    SettleOutcome        SettleOutcome
    OmissionP99Serviced  time.Duration
    OmissionBudgetMs     int      // 0 means auto-scale to 25% of MeasuredP99
    MeasuredP99          time.Duration
    AbortWatcherDeafened bool
    PeakRPSTimesSustain  int
    AbortWindowCap       int
    LivenessFailures     int
    LivenessWatcherTripped bool
    RAWPollIntervalCoarse bool
}

type RunQualityVerdict struct {
    Verdict string   // TRUSTED | DEGRADED | UNTRUSTED
    Issues  []string
}

func EvaluateRunQuality(in *RunQualityInputs) RunQualityVerdict {
    var degraded, untrusted []string
    // UNTRUSTED rules
    if in.DrainTimedOut { untrusted = append(untrusted, "async drain timed out → see USAGE.md#drain-timeout") }
    if in.MeasuredDuration < in.AbortP99Sustain {
        untrusted = append(untrusted, fmt.Sprintf("measured window %v shorter than abort-p99-sustain %v → see USAGE.md#measured-too-short", in.MeasuredDuration, in.AbortP99Sustain))
    }
    if in.WarmupErrorRate > 0.20 {
        untrusted = append(untrusted, fmt.Sprintf("warmup error rate %.0f%% > 20%% → see USAGE.md#warmup-errors", in.WarmupErrorRate*100))
    }
    if !in.SettleOutcome.AllSucceeded && len(in.SettleOutcome.Probes) > 0 {
        untrusted = append(untrusted, fmt.Sprintf("settle phase incomplete: %d/%d probes succeeded → see USAGE.md#settle-incomplete", in.SettleOutcome.Succeeded, len(in.SettleOutcome.Probes)))
    }
    if in.AbortWatcherDeafened && in.AbortWindowCap > 0 && in.PeakRPSTimesSustain > 2*in.AbortWindowCap {
        untrusted = append(untrusted, "abort watcher deafened by sample cap (ratio >2×) → see USAGE.md#abort-watcher-deafened")
    }
    // DEGRADED rules (only if not already UNTRUSTED)
    if in.WarmupErrorRate > 0.05 && in.WarmupErrorRate <= 0.20 {
        degraded = append(degraded, fmt.Sprintf("warmup error rate %.0f%% (5-20%%)", in.WarmupErrorRate*100))
    }
    if in.AbortWatcherDeafened && in.AbortWindowCap > 0 && in.PeakRPSTimesSustain <= 2*in.AbortWindowCap {
        degraded = append(degraded, "abort watcher deafened by sample cap (partial)")
    }
    budgetMs := in.OmissionBudgetMs
    if budgetMs == 0 && in.MeasuredP99 > 0 {
        budgetMs = int(in.MeasuredP99.Milliseconds() / 4) // 25% of measured p99
    }
    if budgetMs > 0 && in.OmissionP99Serviced > time.Duration(budgetMs)*time.Millisecond {
        degraded = append(degraded, fmt.Sprintf("omission p99 %v exceeds budget %dms → see USAGE.md#omission-budget", in.OmissionP99Serviced, budgetMs))
    }
    if in.LivenessFailures > 0 && !in.LivenessWatcherTripped {
        degraded = append(degraded, fmt.Sprintf("liveness probe failed %d times without watcher trip", in.LivenessFailures))
    }
    if in.RAWPollIntervalCoarse {
        degraded = append(degraded, "RAW poll-interval too coarse vs measured p50 → see USAGE.md#raw-poll-bias")
    }

    if len(untrusted) > 0 { return RunQualityVerdict{Verdict: "UNTRUSTED", Issues: append(untrusted, degraded...)} }
    if len(degraded) > 0  { return RunQualityVerdict{Verdict: "DEGRADED",  Issues: degraded} }
    return RunQualityVerdict{Verdict: "TRUSTED"}
}

func PrintRunQuality(out io.Writer, v RunQualityVerdict) {
    sep := strings.Repeat("=", 32)
    fmt.Fprintf(out, "%s RUN QUALITY: %s %s\n", sep, v.Verdict, sep)
    for _, iss := range v.Issues { fmt.Fprintf(out, "  - %s\n", iss) }
    fmt.Fprintln(out, strings.Repeat("=", len(sep)*2+18))
}
```
